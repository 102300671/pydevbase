def table():
    for i in range(1, 10):
        for j in range(i, 10):
            print(f"{i}*{j}={i * j:<{len(str(i * 9))}}", end="\t")
        print()


class High:
    def __init__(self, high):
        self.high = high

    def check_high(self):
        if self.high >= 190:
            print("恭喜!您的身高满足篮球运动员的要求")
        else:
            print("很遗憾，您的身高尚未达到篮球运动员的最低要求")


class Sum:
    def __init__(self, n):
        self.n = n

    def calculate_sum(self):
        total = 0
        i = 1
        while i <= self.n:
            total += i
            i += 1
        print(f"sum(1,{self.n}) = {total}")


class Password:
    def __init__(self, password):
        self.password = password

    def check_password(self):
        if self.password == "miqi":
            print("密码正确")
            return True
        else:
            print("密码错误")
            return False


while True:
    print("1.打印九九乘法表")
    print("2.判断身高是否达到篮球运动员的要求")
    print("3.计算1+2+3+...+n的值")
    print("4.判断密码是否正确")
    print("5.退出")
    choice = int(input("请输入您的选择:"))
    if choice == 1:
        table()
    elif choice == 2:
        while True:
            height = int(input("请输入您的身高,单位:cm(退出:0):"))
            if height == 0:
                break
            high = High(height)
            high.check_high()
    elif choice == 3:
        while True:
            n = int(input("请输入一个整数(退出:0):"))
            if n == 0:
                break
            sum_instance = Sum(n)
            sum_instance.calculate_sum()
    elif choice == 4:
        for i in range(3):
            pwd = input("请输入密码,您有三次机会:")
            password_instance = Password(pwd)
            if password_instance.check_password():
                break
            else:
                print(f"您还有{2 - i}次机会")
    elif choice == 5:
        break
    else:
        print("输入错误，请重新输入")
